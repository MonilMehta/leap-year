10 PRINT "ENTER YEAR"
20 INPUT YEAR

30 REM CHECK IF DIVISIBLE BY 4
31 IF YEAR - INT(YEAR/4) * 4 = 0 GOTO 100
32 GOTO 500

100 REM CHECK IF DIVISIBLE BY 100
101 IF YEAR - INT(YEAR/100) * 100 = 0 GOTO 200
102 GOTO 600

200 REM CHECK IF DIVISIBLE BY 400
201 IF YEAR - INT(YEAR/400) * 400 = 0 GOTO 600
202 GOTO 500

500 PRINT "NO, "; YEAR; " IS NOT A LEAP YEAR"
501 GOTO 10

600 PRINT "YES, "; YEAR; " IS A LEAP YEAR"
601 GOTO 10
